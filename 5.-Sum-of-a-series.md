```C
#include<stdio.h>
#define N 2000000

int main()
 double rb=0, rf=0;
 int i;

 for(i=0; i<=N; i++) rf += 1/(i+3.14);
 for(i=N; i>=0; i--) rb += 1/(i+3.14);
 
 printf(" Forward: %.15e \n Backward: %.15e \n", rf, rb); 
 return 1;
} 
```
